59class Attribute
60{
61
62 public:
63 Attribute(json11::Json doc) {
64
65 missing_field = "";
66 values = std::vector<std::string>();
67 min_provided = false;
68 max_provided = false;
69
70 if (! doc["name"].is_string()) {
71 missing_field = "name";
72 return;
73 }
74 name = doc["name"].string_value();
75
76 if (! doc["type"].is_string()) {
77 missing_field = "type";
78 return;
79 }
80 type = doc["type"].string_value();
81
82 if (! doc["count"].is_number()) {
83 missing_field = "count";
84 return;
85 }
86 count = doc["count"].number_value();
87
88 if (doc["min"].is_number()) {
89 min = doc["min"].number_value();
90 min_provided = true;
91 }
92 if (doc["max"].is_number()) {
93 max = doc["max"].number_value();
94 max_provided = true;
95 }
96 if (doc["values"].is_array()) {
97 std::string tmp;
98 for (json11::Json v : doc["values"].array_items()) {
99 tmp = v.string_value();
100 // on double les backslash, en évitant de traiter les backslash déjà doublés
101 boost::replace_all(tmp, "\\\\", "\\");
102 boost::replace_all(tmp, "\\", "\\\\");
103 // On échappe les doubles quotes
104 boost::replace_all(tmp, "\"", "\\\"");
105 values.push_back(tmp);
106 }
107
108 }
109
110 };
111 ~Attribute(){};
112
113 std::string get_missing_field() {return missing_field;}
114 std::string get_name() {return name;}
115 std::string get_type() {return type;}
116 std::vector<std::string> get_values() {return values;}
117 int get_count() {return count;}
118 int get_min() {return min;}
119 int get_max() {return max;}
120
121 json11::Json to_json() const {
122 json11::Json::object res = json11::Json::object {
123 { "attribute", name },
124 { "count", count },
125 { "type", type }
126 };
127
128 if (min_provided) {
129 res["min"] = min;
130 }
131 if (max_provided) {
132 res["max"] = max;
133 }
134
135 if (values.size() != 0) {
136 res["values"] = values;
137 }
138
139 return res;
140 }
141
142
143 private:
148 std::string missing_field;
149
154 std::string name;
159 std::string type;
164 std::vector<std::string> values;
169 int count;
174 double min;
175 bool min_provided;
180 double max;
181 bool max_provided;
182};
