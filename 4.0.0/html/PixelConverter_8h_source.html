194 void convert_line ( T* bufferto, T* bufferfrom ) {
195
196 T defaultAlpha;
197 if (sizeof(T) == 1) {
198 defaultAlpha = (T) 255;
199 } else if (sizeof(T) == 2) {
200 defaultAlpha = (T) 65535;
201 } else {
202 defaultAlpha = (T) 1;
203 }
204
205 /********************** Depuis 1 canal ********************/
206 if ( input_samplesperpixel == 1) {
207 if (output_samplesperpixel == 2) {
208 for ( int i = 0; i < width; i++ ) {
209 bufferto[2*i] = bufferfrom[i];
210 bufferto[2*i+1] = defaultAlpha;
211 }
212 return;
213 }
214 if (output_samplesperpixel == 3) {
215 for ( int i = 0; i < width; i++ ) {
216 bufferto[3*i] = bufferto[3*i+1] = bufferto[3*i+2] = bufferfrom[i];
217 }
218 return;
219 }
220 if (output_samplesperpixel == 4) {
221 for ( int i = 0; i < width; i++ ) {
222 bufferto[4*i] = bufferto[4*i+1] = bufferto[4*i+2] = bufferfrom[i];
223 bufferto[4*i+3] = defaultAlpha;
224 }
225 return;
226 }
227 }
228
229 /********************** Depuis 2 canaux *******************/
230 if ( input_samplesperpixel == 2) {
231 if (output_samplesperpixel == 1) {
232 for ( int i = 0; i < width; i++ ) {
233 bufferto[i] = bufferfrom[2*i];
234 }
235 return;
236 }
237 if (output_samplesperpixel == 3) {
238 for ( int i = 0; i < width; i++ ) {
239 bufferto[3*i] = bufferto[3*i+1] = bufferto[3*i+2] = bufferfrom[2*i];
240 }
241 return;
242 }
243 if (output_samplesperpixel == 4) {
244 for ( int i = 0; i < width; i++ ) {
245 bufferto[4*i] = bufferto[4*i+1] = bufferto[4*i+2] = bufferfrom[2*i];
246 bufferto[4*i+3] = bufferfrom[2*i + 1];
247 }
248 return;
249 }
250 }
251
252 /********************** Depuis 3 canaux *******************/
253 if ( input_samplesperpixel == 3) {
254 if (output_samplesperpixel == 1) {
255 for ( int i = 0; i < width; i++ ) {
256 bufferto[i] = ( T ) (0.2125*bufferfrom[3*i] + 0.7154*bufferfrom[3*i+1] + 0.0721*bufferfrom[3*i+2]);
257 }
258 return;
259 }
260 if (output_samplesperpixel == 2) {
261 for ( int i = 0; i < width; i++ ) {
262 bufferto[2*i] = ( T ) (0.2125*bufferfrom[3*i] + 0.7154*bufferfrom[3*i+1] + 0.0721*bufferfrom[3*i+2]);
263 bufferto[2*i+1] = defaultAlpha;
264 }
265 return;
266 }
267 if (output_samplesperpixel == 4) {
268 for ( int i = 0; i < width; i++ ) {
269 memcpy(bufferto + 4*i, bufferfrom + 3*i, 3 * sizeof(T));
270 bufferto[4*i+3] = defaultAlpha;
271 }
272 return;
273 }
274 }
275
276 /********************** Depuis 4 canaux *******************/
277 if ( input_samplesperpixel == 4) {
278 if (output_samplesperpixel == 1) {
279 for ( int i = 0; i < width; i++ ) {
280 bufferto[i] = ( T ) (0.2125*bufferfrom[4*i] + 0.7154*bufferfrom[4*i+1] + 0.0721*bufferfrom[4*i+2]);
281 }
282 return;
283 }
284 if (output_samplesperpixel == 2) {
285 for ( int i = 0; i < width; i++ ) {
286 bufferto[2*i] = ( T ) (0.2125*bufferfrom[4*i] + 0.7154*bufferfrom[4*i+1] + 0.0721*bufferfrom[4*i+2]);
287 bufferto[2*i+1] = bufferfrom[4*i + 3];
288 }
289 return;
290 }
291 if (output_samplesperpixel == 3) {
292 for ( int i = 0; i < width; i++ ) {
293 memcpy(bufferto + 3*i, bufferfrom + 4*i, 3 * sizeof(T));
294 }
295 return;
296 }
297 }
298 }
