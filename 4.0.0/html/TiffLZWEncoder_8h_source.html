51template <typename T>
52class TiffLZWEncoder : public TiffEncoder {
53 protected:
54 size_t raw_buffer_size;
55 T* raw_buffer;
56
57 virtual void prepare_header() {
58 BOOST_LOG_TRIVIAL(debug) << "TiffLZWEncoder : preparation de l'en-tete";
59 header_size = TiffHeader::header_size(image->get_channels());
60 header = new uint8_t[header_size];
61 if (image->get_channels() == 1)
62 if (sizeof(T) == sizeof(float)) {
63 memcpy(header, TiffHeader::TIFF_HEADER_LZW_FLOAT32_GRAY, header_size);
64 } else {
65 memcpy(header, TiffHeader::TIFF_HEADER_LZW_INT8_GRAY, header_size);
66 }
67 else if (image->get_channels() == 3)
68 memcpy(header, TiffHeader::TIFF_HEADER_LZW_INT8_RGB, header_size);
69 else if (image->get_channels() == 4)
70 memcpy(header, TiffHeader::TIFF_HEADER_LZW_INT8_RGBA, header_size);
71 *((uint32_t*)(header + 18)) = image->get_width();
72 *((uint32_t*)(header + 30)) = image->get_height();
73 *((uint32_t*)(header + 102)) = image->get_height();
74 *((uint32_t*)(header + 114)) = tmp_buffer_size;
75 }
76
77 virtual void prepare_buffer() {
78 BOOST_LOG_TRIVIAL(debug) << "TiffLZWEncoder : preparation du buffer d'image";
79 int linesize = image->get_width() * image->get_channels();
80 raw_buffer = new T[image->get_height() * image->get_width() * image->get_channels()];
81 raw_buffer_size = 0;
82 int lRead = 0;
83 for (; lRead < image->get_height(); lRead++) {
84 image->get_line(raw_buffer + raw_buffer_size, lRead);
85 raw_buffer_size += linesize;
86 }
87 raw_buffer_size *= sizeof(T);
88 LzwCompressor encoder;
89 tmp_buffer = encoder.encode((uint8_t*)raw_buffer, raw_buffer_size, tmp_buffer_size);
90 delete[] raw_buffer;
91 raw_buffer = NULL;
92 }
93
94 public:
95 TiffLZWEncoder(Image* image, bool is_geotiff = false, int nodata = 0) : TiffEncoder(image, -1, is_geotiff, nodata), raw_buffer_size(0), raw_buffer(NULL) {}
96
97 ~TiffLZWEncoder() {
98 }
99};
