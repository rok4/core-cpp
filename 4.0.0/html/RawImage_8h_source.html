56 virtual int get_line ( uint8_t *buffer, int line ) {
57 size_t size;
58 const uint8_t* data = source->get_data ( size );
59 if ( ! data ) {
60 buffer = 0;
61 return 0;
62 }
63 memcpy ( buffer, ( uint8_t* ) &data[line*channels*width],width*channels*sizeof ( uint8_t ) );

65 }
