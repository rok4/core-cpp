296static bool readHeaderFile(std::string imagefilename, int* width, int* height, int* samplesperpixel, int* bitspersample) {
297
298 const char * pch;
299 pch = strrchr ( imagefilename.c_str(),'.' );
300 int basenamelength = pch - imagefilename.c_str();
301
302 char headerfilename[IMAGE_MAX_FILENAME_LENGTH];
303 memset(headerfilename, 0, IMAGE_MAX_FILENAME_LENGTH);
304 memmove(headerfilename, imagefilename.c_str(), basenamelength);
305
306 memcpy(headerfilename+basenamelength, ".hdr", 4);
307
308 std::ifstream header;
309
310 // Tentative d'ouverture du fichier header, avec l'extension en minuscule ou en majuscule
311 header.open ( headerfilename );
312 if ( !header ) {
313 memcpy(headerfilename+basenamelength, ".HDR", 4);
314 header.open ( headerfilename );
315 if ( !header ) {
316 BOOST_LOG_TRIVIAL(error) << "No found header, with extension .hdr or .HDR" ;
317 return false;
318 }
319 }
320
321 std::string str;
322 char key[20];
323 int value = 0;
324 while (! header.eof() ) {
325 std::getline ( header,str );
326
327 if ( std::sscanf ( str.c_str(),"%s %d", key, &value) != 2 ) continue;
328
329 if (strncmp(key, "NCOLS", 5) == 0 || strncmp(key, "ncols", 5) == 0) *width = value;
330 if (strncmp(key, "NROWS", 5) == 0 || strncmp(key, "nrows", 5) == 0) *height = value;
331 if (strncmp(key, "NBITS", 5) == 0 || strncmp(key, "nbits", 5) == 0) *bitspersample = value;
332 if (strncmp(key, "NBANDS", 6) == 0 || strncmp(key, "nbands", 5) == 0) *samplesperpixel = value;
333 }
334
335 return true;
336}

352static bool uncompressedData(uint8_t* compresseddata, int compresseddatasize, uint8_t* rawdata, int rawdatasize) {
353
354 // Initialisation du flux
355 z_stream zstream;
356 zstream.zalloc = Z_NULL;
357 zstream.zfree = Z_NULL;
358 zstream.opaque = Z_NULL;
359 zstream.data_type = Z_BINARY;
360 int zinit;
361 if ( ( zinit=inflateInit ( &zstream ) ) != Z_OK ) {
362 if ( zinit==Z_MEM_ERROR ) BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : pas assez de memoire" ;
363 else if ( zinit==Z_VERSION_ERROR ) BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : versions de zlib incompatibles" ;
364 else if ( zinit==Z_STREAM_ERROR ) BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : parametres invalides" ;
365 else BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : echec" ;
366 return false;
367 }
368
369 zstream.next_in = compresseddata;
370 zstream.avail_in = compresseddatasize;
371 zstream.next_out = rawdata;
372 zstream.avail_out = rawdatasize;
373 // Decompression du flux
374 while ( zstream.avail_in != 0 ) {
375 if ( int err = inflate ( &zstream, Z_SYNC_FLUSH ) ) {
376 if ( err == Z_STREAM_END && zstream.avail_in == 0 ) break; // fin du fichier OK.
377 if ( zstream.avail_out == 0 ) {
378 BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : le buffer de sortie ne devrait pas être trop petit";
379 return false;
380 }
381 BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : probleme deflate decompression " << err ;
382 return false;
383 }
384 }
385
386 // Destruction du flux
387 if ( inflateEnd ( &zstream ) != Z_OK ) {
388 BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : probleme de liberation du flux" ;
389 return false;
390 }
391
392 if (zstream.avail_out != 0) {
393 BOOST_LOG_TRIVIAL(error) << "Decompression DEFLATE : la taille des données décompressée n'est pas celle attendue" ;
394 BOOST_LOG_TRIVIAL(error) << "Il manque " << zstream.avail_out << "octets" ;
395 return false;
396 }
397
398 return true;
399}
