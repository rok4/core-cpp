155  static bool are_dimensions_consistent(double resolution_x, double resolution_y, int w, int h,
156  BoundingBox<double> bounding_box) {
157  // Vérification de la cohérence entre les résolutions et bbox fournies et les dimensions (en pixel) de l'image
158  // Arrondi a la valeur entiere la plus proche
159 
160  if (resolution_x == 0 || resolution_y == 0)
161  return false;
162 
163  int calcWidth = lround((bounding_box.xmax - bounding_box.xmin) / (resolution_x));
164  int calcHeight = lround((bounding_box.ymax - bounding_box.ymin) / (resolution_y));
165 
166  if (calcWidth != w || calcHeight != h) {
167  BOOST_LOG_TRIVIAL(warning) << "height is " << h << " and calculation give " << calcHeight;
168  BOOST_LOG_TRIVIAL(warning) << "width is " << w << " and calculation give " << calcWidth;
169  return false;
170  }
171 
172  return true;
173  }

498  bool compatible(Image* other) {
499 
500  if (crs != NULL && crs->is_define() && other->get_crs() != NULL && other->get_crs()->is_define() &&
501  !crs->cmp_request_code(other->get_crs()->get_request_code())) {
502  BOOST_LOG_TRIVIAL(debug) << "Different CRS";
503  return false;
504  }
505 
506  if (get_channels() != other->get_channels()) {
507  BOOST_LOG_TRIVIAL(debug) << "Different channels";
508  return false;
509  }
510 
511  double epsilon_x = std::min(get_resx(), other->get_resx()) / 1000.;
512  double epsilon_y = std::min(get_resy(), other->get_resy()) / 1000.;
513 
514  if (fabs(get_resx() - other->get_resx()) > epsilon_x) {
515  BOOST_LOG_TRIVIAL(debug) << "Different X resolutions";
516  return false;
517  }
518  if (fabs(get_resy() - other->get_resy()) > epsilon_y) {
519  BOOST_LOG_TRIVIAL(debug) << "Different Y resolutions";
520  return false;
521  }
522 
523  if (fabs(get_phasex() - other->get_phasex()) > 0.001 && fabs(get_phasex() - other->get_phasex()) < 0.999) {
524  BOOST_LOG_TRIVIAL(debug) << "Different X phasis : " << get_phasex() << " and " << other->get_phasex();
525  return false;
526  }
527  if (fabs(get_phasey() - other->get_phasey()) > 0.001 && fabs(get_phasey() - other->get_phasey()) < 0.999) {
528  BOOST_LOG_TRIVIAL(debug) << "Different Y phasis : " << get_phasey() << " and " << other->get_phasey();
529  return false;
530  }
531 
532  return true;
533  }
