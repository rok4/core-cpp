81template <class Decoder>
82class DataSourceDecoder : public DataSource {
83private:
84 DataSource* encoded_data;
85 const uint8_t* decoded_data;
86 size_t decoded_size;
87public:
88 DataSourceDecoder ( DataSource* encoded_data ) : encoded_data ( encoded_data ), decoded_data ( 0 ), decoded_size ( 0 ) {}
89
90 ~DataSourceDecoder() {
91 if ( decoded_data )
92 delete[] decoded_data;
93 delete encoded_data;
94 }
95
96 const uint8_t* get_data ( size_t &size ) {
97 if ( !decoded_data && encoded_data ) {
98 decoded_data = Decoder::decode ( encoded_data, decoded_size );
99 if ( !decoded_data ) {
100 delete encoded_data;
101 encoded_data = 0;
102 }
103 }
104 size = decoded_size;
105 return decoded_data;
106 }
107
108 bool release_data() {
109 if ( encoded_data ) encoded_data->release_data();
110 if ( decoded_data ) delete[] decoded_data;
111 decoded_data = 0;
112 return true;
113 }
114
115 std::string get_type() {
116 return "image/bil";
117 }
118 int get_http_status() {
119 return 200;
120 }
121 std::string get_encoding() {
122 return "";
123 }
124
125 unsigned int get_length() {
126 return decoded_size;
127 }
128};
