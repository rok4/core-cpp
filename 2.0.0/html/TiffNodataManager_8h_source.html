292 template<typename T>
293 TiffNodataManager<T>::TiffNodataManager ( uint16 channels, int* tv, bool touch_edges, int* dv, int* nv, int t ) :
294  max_channels ( channels ), touch_edges ( touch_edges ), tolerance ( t ) {
295 
296  target_value = new T[channels];
297  data_value = new T[channels];
298  nodata_value = new T[channels];
299 
300  for ( int i = 0; i < channels; i++ ) {
301  target_value[i] = ( T ) tv[i];
302  data_value[i] = ( T ) dv[i];
303  nodata_value[i] = ( T ) nv[i];
304  }
305 
306  if ( memcmp ( tv,nv,channels*sizeof ( int ) ) ) {
307  new_nodata_value = true;
308  } else {
309  // La nouvelle valeur de non-donnée est la même que la couleur cible : on ne change pas la couleur de non-donnée
310  new_nodata_value = false;
311  }
312 
313  if ( touch_edges && memcmp ( tv,dv,channels*sizeof ( int ) ) ) {
314  // Pour changer la couleur des données contenant la couleur cible, il faut avoir l'option "touche les bords".
315  // Sinon, par définition, aucun pixel de la couleur cible est à considérer comme de la donnée.
316  remove_target_value = true;
317  } else {
318  // La nouvelle valeur de donnée est la même que la couleur cible : on ne supprime donc pas la couleur cible des données
319  remove_target_value = false;
320  }
321 }

323 template<typename T>
324 bool TiffNodataManager<T>::process_nodata ( char* input_image_path, char* output_image_path, char* output_mask_path ) {
325  if ( ! new_nodata_value && ! remove_target_value && ! output_mask_path ) {
326  BOOST_LOG_TRIVIAL(info) << "Have nothing to do !" ;
327  return true;
328  }
329 
330  FileImage* source_image = FileImage::create_to_read(input_image_path);
331 
332  if ( source_image == NULL ) {
333  BOOST_LOG_TRIVIAL(error) << "Cannot create the input image "<< input_image_path ;
334  return false;
335  }
336 
337  /* On mémorise certaines informations sur l'image en cours de traitement */
338  width = source_image->get_width();
339  height = source_image->get_height();
340 
341  Photometric::ePhotometric photometric = source_image->get_photometric();
342  Compression::eCompression compression = source_image->get_compression();
343  SampleFormat::eSampleFormat sample_format = source_image->get_sample_format();
344  samplesperpixel = source_image->get_channels();
345 
346  if ( samplesperpixel > max_channels ) {
347  BOOST_LOG_TRIVIAL(error) << "The nodata manager is not adapted (samplesperpixel have to be " << max_channels <<
348  " or less) for the image " << input_image_path << " (" << samplesperpixel << ")";
349  return false;
350  }
351 
352  /*************** Chargement de l'image ***************/
353 
354  T *image_buffer = (T*) malloc (width * height * samplesperpixel * sizeof(T));
355  if ( image_buffer == NULL ) {
356  BOOST_LOG_TRIVIAL(error) << "Cannot allocate a buffer of size " << width * height * samplesperpixel * sizeof(T) << " bytes" ;
357  return false;
358  }
359 
360  BOOST_LOG_TRIVIAL(debug) << "We load input image into memory : " << width * height * samplesperpixel * sizeof(T) / 1024 << " kilobytes";
361  for ( int h = 0; h < height; h++ ) {
362  if (source_image->get_line(image_buffer + width*samplesperpixel*h, h) == 0) {
363  BOOST_LOG_TRIVIAL(error) << "Cannot read line " << h ;
364  return false;
365  }
366  }
367 
368  delete source_image;
369 
370  BOOST_LOG_TRIVIAL(debug) << "Premier pixel : " << (int)image_buffer[0] << "," << (int)image_buffer[1] << "," << (int)image_buffer[2];
371 
372  /************* Calcul du masque de données ***********/
373 
374  uint8_t *mask_buffer = new uint8_t[width * height];
375 
376  bool contain_nodata = identify_nodata_pixels ( image_buffer, mask_buffer );
377 
378  /*************** Modification des pixels *************/
379 
380  if ( remove_target_value ) {
381  BOOST_LOG_TRIVIAL(debug) << "The 'target_value' data pixels are replaced by 'data_value' pixels";
382  change_data_value ( image_buffer, mask_buffer );
383  }
384 
385  if ( new_nodata_value ) {
386  BOOST_LOG_TRIVIAL(debug) << "Nodata pixels which touch edges are replaced by 'nodata_value' pixels";
387  change_nodata_value ( image_buffer, mask_buffer );
388  }
389 
390  /**************** Ecriture de l'images ****************/
391 
392  /* Seulement si on a modifié l'image. Si seule l'écriture du masque nous intéressait, on ne réecrit pas l'image,
393  * même si un chemin d'image différent est fourni pour la sortie */
394  if ( remove_target_value || new_nodata_value ) {
395 
396  FileImage* output_image = FileImage::create_to_write(
397  output_image_path, BoundingBox<double>(0,0,0,0), -1, -1, width, height,
398  samplesperpixel, sample_format, photometric, compression
399  );
400 
401  if ( output_image == NULL ) {
402  BOOST_LOG_TRIVIAL(error) << "Cannot create the output image "<< output_image_path ;
403  return false;
404  }
405 
406  BOOST_LOG_TRIVIAL(debug) << "We write the output image " << output_image_path;
407  output_image->write_image(image_buffer);
408 
409  delete output_image;
410 
411  } else {
412  if ( memcmp ( input_image_path, output_image_path, sizeof ( output_image_path ) ) )
413  BOOST_LOG_TRIVIAL(info) << "The image have not be modified, the file '" << output_image_path <<"' is not written" ;
414  }
415 
416  /**************** Ecriture du masque ? ****************/
417  if ( output_mask_path ) {
418  if (contain_nodata) {
419  FileImage* output_mask = FileImage::create_to_write(
420  output_mask_path, BoundingBox<double>(0,0,0,0), -1, -1, width, height,
421  1, SampleFormat::UINT8, Photometric::MASK, Compression::DEFLATE
422  );
423 
424  if ( output_mask == NULL ) {
425  BOOST_LOG_TRIVIAL(error) << "Cannot create the output mask "<< output_mask_path ;
426  return false;
427  }
428 
429  BOOST_LOG_TRIVIAL(debug) << "We write the output mask " << output_mask_path;
430  output_mask->write_image(mask_buffer);
431 
432  delete output_mask;
433  } else {
434  BOOST_LOG_TRIVIAL(info) << "The image contains only data, the mask '" << output_mask_path <<"' is not written" ;
435  }
436  }
437 
438  free(image_buffer);
439  delete[] mask_buffer;
440 
441  return true;
442 }

444 template<typename T>
445 inline bool TiffNodataManager<T>::is_target_value ( T* pix ) {
446  int pixint;
447  int tvint;
448  for ( int i = 0; i < samplesperpixel; i++ ) {
449  pixint = ( int ) pix[i];
450  tvint = ( int ) target_value[i];
451  if ( pixint < tvint - tolerance || pixint > tvint + tolerance ) {
452  return false;
453  }
454  }
455  return true;
456 }

458 template<typename T>
459 bool TiffNodataManager<T>::identify_nodata_pixels ( T* IM, uint8_t* MSK ) {
460 
461  BOOST_LOG_TRIVIAL(debug) << "Identify nodata pixels..." ;
462  memset ( MSK, 255, width * height );
463 
464  bool containNodata = false;
465 
466  if ( touch_edges ) {
467  BOOST_LOG_TRIVIAL(debug) << "\t...which touch edges" ;
468  // On utilise la couleur target_value et on part des bords
469  queue<unsigned long long> Q;
470 
471  // Initialisation : we identify front pixels which are 'target_value'
472  for ( unsigned long long pos = 0; pos < width; pos++ ) { // top
473  if ( is_target_value ( IM + samplesperpixel * pos ) ) {
474  Q.push ( pos );
475  MSK[pos] = 0;
476  }
477  }
478  for ( unsigned long long pos = width* ( height-1 ); pos < width*height; pos++ ) { // bottom
479  if ( is_target_value ( IM + samplesperpixel * pos ) ) {
480  Q.push ( pos );
481  MSK[pos] = 0;
482  }
483  }
484  for ( unsigned long long pos = 0; pos < width*height; pos += width ) { // left
485  if ( is_target_value ( IM + samplesperpixel * pos ) ) {
486  Q.push ( pos );
487  MSK[pos] = 0;
488  }
489  }
490  for ( unsigned long long pos = width -1; pos < width*height; pos+= width ) { // right
491  if ( is_target_value ( IM + samplesperpixel * pos ) ) {
492  Q.push ( pos );
493  MSK[pos] = 0;
494  }
495  }
496 
497  if ( Q.empty() ) {
498  // No nodata pixel identified, nothing to do
499  return false;
500  }
501 
502  containNodata = true;
503 
504  // while there are 'target_value' pixels which can propagate, we do it
505  while ( !Q.empty() ) {
506  unsigned long long pos = Q.front();
507  Q.pop();
508  unsigned long long newpos;
509  if ( pos % width > 0 ) {
510  newpos = pos - 1;
511  if ( MSK[newpos] && is_target_value ( IM + newpos*samplesperpixel ) ) {
512  MSK[newpos] = 0;
513  Q.push ( newpos );
514  }
515  }
516  if ( pos % width < width - 1 ) {
517  newpos = pos + 1;
518  if ( MSK[newpos] && is_target_value ( IM + newpos*samplesperpixel ) ) {
519  MSK[newpos] = 0;
520  Q.push ( newpos );
521  }
522  }
523  if ( pos / width > 0 ) {
524  newpos = pos - width;
525  if ( MSK[newpos] && is_target_value ( IM + newpos*samplesperpixel ) ) {
526  MSK[newpos] = 0;
527  Q.push ( newpos );
528  }
529  }
530  if ( pos / width < height - 1 ) {
531  newpos = pos + width;
532  if ( MSK[newpos] && is_target_value ( IM + newpos*samplesperpixel ) ) {
533  MSK[newpos] = 0;
534  Q.push ( newpos );
535  }
536  }
537  }
538 
539  } else {
540  BOOST_LOG_TRIVIAL(debug) << "\t..., all pixels in 'target color'" ;
541  // Tous les pixels de la couleur target_value sont à considérer comme du nodata
542 
543  for ( unsigned long long i = 0; i < width * height; i++ ) {
544 
545  if ( is_target_value ( IM+i*samplesperpixel ) ) {
546  containNodata = true;
547  MSK[i] = 0;
548  }
549  }
550 
551  }
552 
553  return containNodata;
554 }
