292 template<typename T>
293 void Line::write ( T* buffer, int output_channels) {
294  switch ( output_channels ) {
295  case 1:
296  for ( int i = 0; i < width; i++ ) {
297  buffer[i] = ( T ) ( 0.2125*samples[3*i] + 0.7154*samples[3*i+1] + 0.0721*samples[3*i+2] ) * alpha[i];
298  }
299  break;
300  case 2:
301  for ( int i = 0; i < width; i++ ) {
302  buffer[2*i] = ( T ) ( 0.2125*samples[3*i] + 0.7154*samples[3*i+1] + 0.0721*samples[3*i+2] );
303  buffer[2*i+1] = ( T ) ( alpha[i]*coeff );
304  }
305  break;
306  case 3:
307  for ( int i = 0; i < width; i++ ) {
308  buffer[3*i] = ( T ) (alpha[i] * samples[3*i]);
309  buffer[3*i+1] = ( T ) (alpha[i] * samples[3*i+1]);
310  buffer[3*i+2] = ( T ) (alpha[i] * samples[3*i+2]);
311  }
312  break;
313  case 4:
314  for ( int i = 0; i < width; i++ ) {
315  buffer[4*i] = ( T ) (samples[3*i]);
316  buffer[4*i+1] = ( T ) (samples[3*i+1]);
317  buffer[4*i+2] = ( T ) (samples[3*i+2]);
318  buffer[4*i+3] = ( T ) ( alpha[i]*coeff );
319  }
320  break;
321  }
322 }
