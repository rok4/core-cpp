158  void calculate_offsets() {
159  rows_offsets.clear();
160  c0s.clear();
161  c1s.clear();
162  c2s.clear();
163 
164  for ( int i = 0; i < ( int ) source_images.size(); i++ ) {
165 
166  double y = source_images[i]->line_to_y ( 0 );
167 
168  rows_offsets.push_back(y_to_line ( y ));
169  c0s.push_back(std::max ( 0,x_to_column ( source_images[i]->get_xmin() + 0.5*source_images[i]->get_resx() ) ));
170  c1s.push_back(std::min ( width - 1,x_to_column ( source_images[i]->get_xmax() - 0.5*source_images[i]->get_resx() ) ));
171  c2s.push_back(std::max ( 0, source_images[i]->x_to_column ( bbox.xmin + 0.5*resx ) ) );
172  }
173  }
