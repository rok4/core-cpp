239  void close_connection() {
240  connected = false;
241  // On met à jour le fichier de jeton d'authentification Swift s'il a été fourni et que le token utilisé n'est pas celui dedans
242  if(token_file != "" && ! use_token_from_file) {
243  std::fstream token_stream;
244  token_stream.open(token_file, std::ios::out);
245  if ( token_stream.is_open() ) {
246  token_stream << token;
247  token_stream.close();
248  BOOST_LOG_TRIVIAL(debug) << "Token file " << token_file << " updated with new token " << token ;
249  } else {
250  BOOST_LOG_TRIVIAL(warning) << "Token file " << token_file << " could not be updated with new token " << token;
251  }
252  }
253  else if(token_file != "") {
254  BOOST_LOG_TRIVIAL(debug) << "Token file " << token_file << " already contains the used token " << token ;
255  }
256  }
