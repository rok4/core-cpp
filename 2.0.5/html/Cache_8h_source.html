542  static void add_slab_infos(std::string origin_slab_name, Context* data_context, std::string data_slab_name, int tiles_number, uint8_t* offsets, uint8_t* sizes) {
543  // On utilise le nom original de la dalle (celle à lire a priori) comme clé dans la map
544  // Potentiellement ce nom est différent de la vraie dalle contenant la donnée (dans le cas d'une dalle symbolique)
545  // mais c'est via cette dalle symbolique que la donnée est a priori requêtée
546  // donc c'est ce nom qu'on utilisera pour l'interrogation du cache
547 
548  mtx.lock();
549 
550  IndexElement* elem = new IndexElement(origin_slab_name, data_context, data_slab_name, tiles_number, offsets, sizes);
551 
552  // L'information n'est a priori pas dans le cache car
553  // Soit elle était dans le cache et valide, alors on aurait utilisé ce cache et on n'aurait pas fait appel à cet ajout
554  // Soit elle était dans le cache mais obsolète, alors on l'aurait déjà supprimé du cache
555 
556  if (cache.size() == size) {
557  // On récupère le dernier élément du cache
558  IndexElement* last = cache.back();
559  // On le vire du cache
560  cache.pop_back();
561  // On le déréférence de la map d'accès
562  map.erase(last->key);
563  delete last;
564  }
565 
566  // update reference
567  cache.push_front(elem);
568  map[origin_slab_name] = cache.begin();
569 
570  mtx.unlock();
571  };
572 
590  static bool get_slab_infos(std::string key, int tile_number, Context** data_context, std::string* data_slab_name, uint32_t* offset, uint32_t* size) {
591  std::unordered_map<std::string, std::list<IndexElement *>::iterator>::iterator it = map.find ( key );
592  if ( it == map.end() ) {
593  return false;
594  } else {
595  // Gestion de la péremption du cache (une heure max)
596  std::time_t now = std::time(NULL);
597  if (now - (*(it->second))->date > validity) {
598  mtx.lock();
599 
600  // on le cherche à nouveau pour vérifier qu'il n'a pas déjà été supprimé par un thread concurrent
601  it = map.find ( key );
602 
603  if ( it != map.end() ) {
604  delete *(it->second);
605  cache.erase(it->second);
606  map.erase(it);
607  }
608 
609  mtx.unlock();
610 
611  return false;
612  }
613 
614  *data_slab_name = (*(it->second))->name;
615  *data_context = (*(it->second))->context;
616  *offset = (*(it->second))->offsets.at(tile_number);
617  *size = (*(it->second))->sizes.at(tile_number);
618 
619  // On fait le choix de ne pas remettre en tête du cache cet élément, même s'il vient d'être accéder
620  // C'est parce qu'en plus d'avoir une taille limite, les élément cachés ont une date de péromption
621  // Il serait donc dommage de remettre en avant (donc plus loin d'une suppression par taille de cache atteinte)
622  // un élément qui va de toute manière finir par être obsolète.
623  // C'est une différence avec une cache LRU classique
624 
625  return true;
626  }
627  };
