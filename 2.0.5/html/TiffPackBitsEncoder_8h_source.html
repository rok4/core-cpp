63  virtual void prepare_header(){
64  BOOST_LOG_TRIVIAL(debug) << "TiffPackBitsEncoder : preparation de l'en-tete";
65  header_size = TiffHeader::header_size ( image->get_channels() );
66  header = new uint8_t[header_size];
67  if ( image->get_channels()==1 )
68  if ( sizeof ( T ) == sizeof ( float ) ) {
69  memcpy( header, TiffHeader::TIFF_HEADER_PKB_FLOAT32_GRAY, header_size);
70  } else {
71  memcpy( header, TiffHeader::TIFF_HEADER_PKB_INT8_GRAY, header_size);
72  }
73  else if ( image->get_channels()==3 )
74  memcpy( header, TiffHeader::TIFF_HEADER_PKB_INT8_RGB, header_size);
75  else if ( image->get_channels()==4 )
76  memcpy( header, TiffHeader::TIFF_HEADER_PKB_INT8_RGBA, header_size);
77  * ( ( uint32_t* ) ( header+18 ) ) = image->get_width();

83  virtual void prepare_buffer(){
84  BOOST_LOG_TRIVIAL(debug) << "TiffPackBitsEncoder : preparation du buffer d'image";
85  int linesize = image->get_width()*image->get_channels();
86  tmp_buffer = new uint8_t[linesize* image->get_height() * sizeof ( T ) *2];
87  tmp_buffer_size = 0;
88  raw_buffer = new T[linesize];
89  raw_buffer_size = linesize * sizeof ( T );
90  int lRead = 0;
91  PkbCompressor encoder;
92  uint8_t * pkbLine;
93  for ( ; lRead < image->get_height() ; lRead++ ) {
94  image->get_line ( raw_buffer, lRead );
95  size_t pkbLineSize = 0;
96  pkbLine = encoder.encode ( ( uint8_t* ) raw_buffer,raw_buffer_size, pkbLineSize );
97  memcpy ( tmp_buffer+tmp_buffer_size,pkbLine,pkbLineSize );
98  tmp_buffer_size += pkbLineSize;
99  delete[] pkbLine;
100  }
101  delete[] raw_buffer;
102  raw_buffer = NULL;
103  }
