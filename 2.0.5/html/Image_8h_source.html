158  static bool are_dimensions_consistent(double resolution_x, double resolution_y, int w, int h, BoundingBox<double> bounding_box) {
159  // Vérification de la cohérence entre les résolutions et bbox fournies et les dimensions (en pixel) de l'image
160  // Arrondi a la valeur entiere la plus proche
161 
162  if (resolution_x == 0 || resolution_y == 0) return false;
163 
164  int calcWidth = lround ( ( bounding_box.xmax - bounding_box.xmin ) / ( resolution_x ) );
165  int calcHeight = lround ( ( bounding_box.ymax - bounding_box.ymin ) / ( resolution_y ) );
166 
167  if ( calcWidth != w || calcHeight != h ) {
168  BOOST_LOG_TRIVIAL(warning) << "height is " << h << " and calculation give " << calcHeight ;
169  BOOST_LOG_TRIVIAL(warning) << "width is " << w << " and calculation give " << calcWidth ;
170  return false;
171  }
172 
173  return true;
174  }

246  inline bool set_dimensions ( int w, int h, BoundingBox<double> box, double rx, double ry ) {

523  bool compatible ( Image* other ) {
524 
525  if ( crs != NULL && crs->is_define() && other->get_crs() != NULL && other->get_crs()->is_define() && ! crs->cmp_request_code(other->get_crs()->get_request_code()) ) {
526  BOOST_LOG_TRIVIAL(debug) << "Different CRS" ;
527  return false;
528  }
529 
530  if ( get_channels() != other->get_channels() ) {
531  BOOST_LOG_TRIVIAL(debug) << "Different channels" ;
532  return false;
533  }
534 
535  double epsilon_x=std::min ( get_resx(), other->get_resx() ) /1000.;
536  double epsilon_y=std::min ( get_resy(), other->get_resy() ) /1000.;
537 
538  if ( fabs ( get_resx()-other->get_resx() ) > epsilon_x ) {
539  BOOST_LOG_TRIVIAL(debug) << "Different X resolutions" ;
540  return false;
541  }
542  if ( fabs ( get_resy()-other->get_resy() ) > epsilon_y ) {
543  BOOST_LOG_TRIVIAL(debug) << "Different Y resolutions" ;
544  return false;
545  }
546 
547  if ( fabs ( get_phasex()-other->get_phasex() ) > 0.001 && fabs ( get_phasex()-other->get_phasex() ) < 0.999 ) {
548  BOOST_LOG_TRIVIAL(debug) << "Different X phasis : " << get_phasex() << " and " << other->get_phasex() ;
549  return false;
550  }
551  if ( fabs ( get_phasey()-other->get_phasey() ) > 0.001 && fabs ( get_phasey()-other->get_phasey() ) < 0.999 ) {
552  BOOST_LOG_TRIVIAL(debug) << "Different Y phasis : " << get_phasey() << " and " << other->get_phasey() ;
553  return false;
554  }
555 
556  return true;
557  }
