162  template<typename T>
163  void unassociate_alpha ( T* buffer ) {
164 
165  T coeff;
166  if (sizeof(T) == 1) {
167  // pour les entiers sur 8 bits
168  coeff = (T) 255;
169  } else if (sizeof(T) == 2) {
170  // pour les entiers sur 16 bits
171  coeff = (T) 65535;
172  } else {
173  coeff = (T) 1;
174  }
175 
176  T* pix = buffer;
177  for (int i = 0; i < width; i++, pix += channels) {
178  T alpha = *(pix + channels - 1);
179 
180  if (alpha == coeff) {
181  // Opacité pleine
182  continue;
183  }
184  if (alpha == 0) {
185  // Transparence complète
186  memset(pix, 0, channels);
187  continue;
188  }
189 
190  for (int c = 0; c < channels - 1; c++) {
191  pix[c] = pix[c] * coeff / alpha;
192  }
193  }
194  }
