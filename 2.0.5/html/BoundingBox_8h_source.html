300  void phase(BoundingBox<T> other, T resx, T resy) {
301 
302  double intpart;
303  double phi = 0;
304  double phaseDiff = 0;
305 
306  // phase de la bbox fournie, de référence, sur laquelle se caser
307  // la bbox en entrée est en accord avec les résolutions fournies,
308  // c'est à dire que la phase est la même pour le min et le max
309  // ce n'est pas forcément le cas pour la bbox à mettre en phase
310  double phaseX = other.get_xmin_phase(resx);
311  double phaseY = other.get_ymin_phase(resy);
312 
313  // Mise en phase de xmin (sans que celui ci puisse être plus petit)
314  phi = get_xmin_phase(resx);
315 
316  if ( fabs ( phi-phaseX ) > 0.0001 && fabs ( phi-phaseX ) < 0.9999 ) {
317  phaseDiff = phaseX - phi;
318  if ( phaseDiff < 0. ) {
319  phaseDiff += 1.0;
320  }
321  xmin += phaseDiff * resx;
322  }
323 
324  // Mise en phase de xmax (sans que celui ci puisse être plus grand)
325  phi = get_xmax_phase(resx);
326 
327  if ( fabs ( phi-phaseX ) > 0.0001 && fabs ( phi-phaseX ) < 0.9999 ) {
328  phaseDiff = phaseX - phi;
329  if ( phaseDiff > 0. ) {
330  phaseDiff -= 1.0;
331  }
332  xmax += phaseDiff*resx;
333  }
334 
335  // Mise en phase de ymin (sans que celui ci puisse être plus petit)
336  phi = get_ymin_phase(resy);
337 
338  if ( fabs ( phi-phaseY ) > 0.0001 && fabs ( phi-phaseY ) < 0.9999 ) {
339  phaseDiff = phaseY - phi;
340  if ( phaseDiff < 0. ) {
341  phaseDiff += 1.0;
342  }
343  ymin += phaseDiff*resy;
344  }
345 
346  // Mise en phase de ymax (sans que celui ci puisse être plus grand)
347  phi = get_ymax_phase(resy);
348 
349  if ( fabs ( phi-phaseY ) > 0.0001 && fabs ( phi-phaseY ) < 0.9999 ) {
350  phaseDiff = phaseY - phi;
351  if ( phaseDiff > 0. ) {
352  phaseDiff -= 1.0;
353  }
354  ymax += phaseDiff*resy;
355  }
356 
357  }
